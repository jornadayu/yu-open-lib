# useViewport

```tsx
import { useViewport } from 'yu-open-lib'

const MyComponent: React.FC = () => {
  const { isMobile } = useViewport()

  return (
    <Card>
      isMobile: {isMobile.toString()}

      {isMobile && <div>I only appear on mobile viewports</div>}
    </Card>
  )
}
```
