### AppTheme Usage

```tsx
import React from 'react'

import { ThemeProvider } from '@mui/material/styles'

import 'yu-open-lib/dist/index.css'
import { AppTheme } from 'yu-open-lib'

const theme = AppTheme({ darkMode: false })

const App = () => {
  return (
    <ThemeProvider theme={theme}>
      ...
    </ThemeProvider>
  )
}
```
